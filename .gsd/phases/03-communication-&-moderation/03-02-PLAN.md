<task type="auto">
  <name>Create POST /api/messages endpoint for sending messages</name>
  <files>src/app/api/messages/route.ts</files>
  <action>
  Create POST /api/messages endpoint:
  - Authenticate user with auth()
  - Accept JSON body: {matchId: string, content: string}
  - Validate matchId exists and current user is participant (user1Id or user2Id)
  - Validate content is non-empty and <= 5000 chars
  - Create Message record with prisma.message.create()
  - Fields: matchId, senderId (current user), content, isRead: false
  - Return 201 with created message object
  - Return 401 if not authenticated, 403 if not match participant, 400 for validation errors
  
  Follow patterns from src/app/api/likes/route.ts for authentication and error handling.
  </action>
  <verify>curl -X POST /api/messages with valid token returns 201 with message object</verify>
  <done>POST /api/messages creates messages in database for authenticated match participants</done>
</task>

<task type="auto">
  <name>Create GET /api/messages/[matchId] endpoint for message history</name>
  <files>src/app/api/messages/[matchId]/route.ts</files>
  <action>
  Create GET /api/messages/[matchId] endpoint:
  - Authenticate user with auth()
  - Extract matchId from URL params
  - Validate match exists and current user is participant (user1Id or user2Id)
  - Query messages with prisma.message.findMany() where matchId equals param
  - Include sender relation (select: id, name, photoUrls)
  - Order by createdAt ASC (chronological)
  - Return 200 with {messages: Message[]} array
  - Return 401 if not authenticated, 403 if not match participant, 404 if match not found
  
  Follow patterns from src/app/api/discovery/route.ts for includes and ordering.
  </action>
  <verify>curl GET /api/messages/[valid-match-id] returns 200 with message array</verify>
  <done>GET /api/messages/[matchId] returns message history for authenticated match participants</done>
</task>

<task type="auto">
  <name>Create GET /api/socket SSE endpoint for real-time updates</name>
  <files>src/app/api/socket/route.ts</files>
  <action>
  Create GET /api/socket endpoint using Server-Sent Events:
  - Set headers: Content-Type: text/event-stream, Cache-Control: no-cache, Connection: keep-alive
  - Authenticate user with auth()
  - Create TextEncoder for streaming
  - Create ReadableStream that:
    - Sends heartbeat ping every 30 seconds to keep connection alive
    - Listens for new messages where user is match participant
    - Sends SSE event when new message: data: {type: 'message', matchId, message}\n\n
  - Return NextResponse with stream body
  
  For MVP, use simple polling approach: query database every 2 seconds for new messages since last check.
  More sophisticated pub/sub can be added later if needed.
  
  Follow Next.js App Router streaming patterns from official docs.
  </action>
  <verify>curl GET /api/socket streams events, heartbeat pings received every 30s</verify>
  <done>SSE endpoint streams real-time message updates to authenticated users</done>
</task>
